# Instalasi Di Termux / Linux
- $pkg update && pkg upgrade
- $pkg install git -y
- $pkg install python -y
- $pip install json
- $pip install requests
- $git clone https://github.com/putragilanq/GitGet
- $cd GitGet
- $python get.py
